package extractor

import (
	"fmt"
	"os/exec"
)

func CallPythonExtractor() {

	// Ejecuta el script de Python y captura la salida
	cmd := exec.Command("python3", "readMap.py")
	out, err := cmd.Output()
	if err != nil {
		fmt.Println("Error al ejecutar el script de Python:", err)
		return
	}

	// La variable 'out' contiene la salida del script de Python como un slice de bytes
	fmt.Println("Salida del script de Python:", string(out))

	// Aquí puedes procesar la salida y, por ejemplo, asignarla a una variable
	valorExtraido := string(out)

	// Utiliza 'valorExtraido' como necesites en tu script de Go
	fmt.Println("Valor extraído:", valorExtraido)

}
